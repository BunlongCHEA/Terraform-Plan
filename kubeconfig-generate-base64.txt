#!/bin/bash

# SSH to your K3s server
ssh root@206.189.156.230

# Once connected, run these commands:

# Get the kubeconfig
sudo cat /etc/rancher/k3s/k3s.yaml > /tmp/k3s-config.yaml

# Replace 127.0.0.1 with your public IP
sed -i 's/127.0.0.1/206.189.156.230/g' /tmp/k3s-config.yaml

# Display the kubeconfig (to verify)
cat /tmp/k3s-config.yaml

# Convert to base64 (single line, no line breaks)
cat /tmp/k3s-config.yaml | base64 -w 0 > /tmp/k3s-config-base64.txt

# Display base64 encoded kubeconfig
cat /tmp/k3s-config-base64.txt
