Step 1: Get Cloudflare Origin Certificate
- Go to Cloudflare Dashboard
- Select your domain bunlong.uk
- Go to SSL/TLS → Origin Server
- Click Create Certificate
- Select:
    Let Cloudflare generate a private key and CSR
    Hostnames: *.bunlong.uk, bunlong.uk
    Validity: 15 years
- Click Create
- Copy both the certificate and private key


Step 2: Delete old certificate first
# Delete pods in Argocd first to avoid Retry to created

# Delete the old certificate secret
kubectl delete secret kyc-blockchain-tls -n kyc-blockchain

# Verify it's deleted
kubectl get secret kyc-blockchain-tls -n kyc-blockchain


Step 3: Create Secret from Cloudflare Certificate
# Save certificate to file
cat > cloudflare-cert.pem << 'EOF'
-----BEGIN CERTIFICATE-----
[PASTE YOUR CLOUDFLARE CERTIFICATE HERE]
-----END CERTIFICATE-----
EOF

# Save private key to file
cat > cloudflare-key.pem << 'EOF'
-----BEGIN PRIVATE KEY-----
[PASTE YOUR CLOUDFLARE PRIVATE KEY HERE]
-----END PRIVATE KEY-----
EOF

# Create Kubernetes secret
kubectl create secret tls kyc-blockchain-tls \
  --cert=cloudflare-cert.pem \
  --key=cloudflare-key.pem \
  -n kyc-blockchain

# Verify
kubectl describe secret kyc-blockchain-tls -n kyc-blockchain


Step 4: Extract Certificate from Kubernetes Secret
# Extract certificate
kubectl get secret kyc-blockchain-tls -n kyc-blockchain -o jsonpath='{.data.tls\.crt}' | base64 -d > tls.crt

# Extract private key
kubectl get secret kyc-blockchain-tls -n kyc-blockchain -o jsonpath='{.data.tls\.key}' | base64 -d > tls.key

# Verify files were created
ls -lh tls.crt tls.key

# View certificate info
openssl x509 -in tls.crt -noout -text | head -20

## Encode certificate (no line wrapping)
cat tls.crt | base64 -w 0 > tls.crt.b64

# Encode private key (no line wrapping)
cat tls.key | base64 -w 0 > tls.key.b64

# The view to Copy
cat tls.crt.b64
